string to_lower(const string& s, const int ini) {
   string ret;
   termino=false;
   for(int i = ini; i < (int)s.size(); i++){
        if(isalpha(s[i])){
            ret.push_back(tolower(s[i]));
        } else{
            fin=i+1;
            return ret;
        }
    }
    termino=true;
   return ret;
}
